15 /*
16  * tabulation.h. This file defines the Tabulation object, which keeps
17  * track of comparisons between pairs of SNPs. These comparisons take
18  * place within a window of size windocm centimorgans, which slides
19  * across the genome. Comparisons are tabulated into nbins bins, based
20  * on the distance that separates them along the chromosome. The
21  * Tabulation object represents a table, with a row for each bin, and
22  * a column for each of: nobs, sep_cm, numerator, denominator. Each
23  * time a pair of SNPs is compared, we add something to each of these
24  * columns in the row determined by the distance that separates the
25  * SNPs along the chromosome.
26  *
27  * The definition of struct Tabulation is exposed here (rather than
28  * hidden in tabulation.c) for the benefit of the inline functions
29  * defined at the bottom of this file.
30  */
31 struct Tabulation {
32  int nbins; /* how many bins to use in tabulating results */
33  double windowcm; /* width of sliding window in centimorgans */
34  double bins_per_cm; /* nbins / windowcm */
35  long unsigned *nobs; /* number of observations w/i each bin */
36  double *sep_cm; /* summed separation of SNPs, by bin */
37  double *numerator; /* summed D^2 values */

39  double *rsq; /* summed values of r^2 */

44  */

46 };

89  * Record a single comparison in the appropriate entry of the arrays.
90  * Wgt is the number of copies of this comparison. May be 0 or >1 in
91  * bootstrap samples.
92  */
93 static inline void Tabulation_record(Tabulation * tab, double Dsq,
94  double pqpq, double sep_cm,
95  unsigned wgt) {
96 
97  myassert(Dsq >= 0.0);
98  myassert(pqpq > 0.0);
99 
100  /* determine bin from sep_cm */
101  register int ndx = getBin(sep_cm, tab->bins_per_cm);
102 
103 #ifndef NDEBUG
104  if(ndx < 0 || ndx >= tab->nbins) {
105  fprintf(stderr, "ERR @%s:%d: ndx=%d; not in [0,%d].\n",
106  __FILE__, __LINE__, ndx, tab->nbins - 1);
107 
108  if(sep_cm < 0 || sep_cm >= tab->windowcm)
109  fprintf(stderr, "Cause: sep_cm=%1.15lg is not in [0,%1.15lg)?\n",
110  sep_cm, tab->windowcm);
111  else
112  fprintf(stderr, "However, sep_cm=%1.15lg IS in [0,%1.15lg)?\n",
113  sep_cm, tab->windowcm);
114  exit(EXIT_FAILURE);
115  }
116 #endif
117 
118  tab->nobs[ndx] += wgt;
119  tab->numerator[ndx] += wgt * Dsq;
120  tab->denominator[ndx] += wgt * pqpq;
121  tab->rsq[ndx] += wgt * Dsq / pqpq;
122  tab->sep_cm[ndx] += wgt * sep_cm;
